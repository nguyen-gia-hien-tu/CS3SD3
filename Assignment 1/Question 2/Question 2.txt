HOTEL = START,

REQUESTED = (send_confirmation -> CONFIRMED
	    |reserve -> WAITING),
CONFIRMED = (move_in -> pay -> ARCHIVED
	    |cancel -> STOP),
WAITING = (send_confirmation -> CONFIRMED
	  |cancel -> STOP),
ARCHIVED = (store_record -> STOP).
