/*
FSP for Question 9 using LTSA syntax style
*/

COMP1 = (idle1 -> read1 -> COMP1
	|idle1 -> write1 -> COMP1).
COMP2 = (idle2 -> read2 -> COMP2
	|idle2 -> write2 -> COMP2).
MUT = (write1 -> MUT
      |write2 -> MUT).
||SHARE = (COMP1 || COMP2 || MUT).
